@startuml

class ClienteService {
    -repo: ClienteRepository
    +ClienteService()
    +obtenerClientes()
    +buscarPorMatricula(matricula: String): Cliente
    +esFrecuente(cliente: Cliente): boolean
}

class MembresiaService {
    +estaVencida(membresia: Membresia): boolean
    +cambiarMembresia(cliente: Cliente, nueva: TipoMembresia): void
    +obtenerDescripcion(tipo: TipoMembresia): String
}

class PagoService {
    +registrarPago(cliente: Cliente, pago: Pago): void
    +aplicarDescuento(cliente: Cliente, monto: double): double
}

class AccesoService {
    -repo: HistorialRepository
    +AccesoService()
    +permitirAcceso(cliente: Cliente): boolean
    -registrarAcceso(cliente: Cliente): void
}

class ClaseService {
    +inscribirCliente(cliente: Cliente, clase: Clase): boolean
    +cancelarAsistencia()
}

class RenovacionService {
    +activar(cliente: Cliente, dia: String)
    +desactivar(cliente: Cliente)
}

class ReporteService {
    +generarReporteClientes(clientes: Lista<Cliente>): void
}

ClienteService --> ClienteRepository
AccesoService --> HistorialRepository
ReporteService --> Cliente
MembresiaService --> Cliente
MembresiaService --> TipoMembresia
PagoService --> Cliente
PagoService --> Pago
ClaseService --> Cliente
ClaseService --> Clase

@enduml
